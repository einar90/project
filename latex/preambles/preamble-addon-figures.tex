\usepackage{etex}
\usepackage{tikz,pgfplots}
\pgfplotsset{compat=1.9}

                           at={(0.5,-0.20)},
                           anchor=north,
                           legend cell align=left,
                           legend columns=3,
                           font=\sffamily,
                           column sep=0.3cm}}
\pgfplotsset{width=20cm, height=12cm}
\pgfplotsset{
every axis label/.append style={font=\sffamily},
tick label style={font=\sffamily},
}
